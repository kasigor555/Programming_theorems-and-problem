# Задачи без массивов

#### **1.1.8.** 
Вычисление факториала. Дано натуральное **n**, вычислите **n!**

>**Решение 1** Если n = 0, вернуть 1 (0! = 1), иначе n! = n · (n − 1)!

```javascript

function fact(n) {

  let count = 1;
  let res = 1;
  
  while (count <= n){
    res = res * count;
    count += 1;
  }
  
  return res;
}

console.log(fact(2));
console.log(fact(0));
console.log(fact(5));
console.log(fact(-5));

```

>**Решение 2** Через рекурсию (рекурсивный процесс).
```javascript

function fact(n) {

  if (n <= 1) {
    return 1;
  } 
  
  return n * fact(n - 1);
}

console.log(fact(2));
console.log(fact(0));
console.log(fact(5));
console.log(fact(-5));

```

>**Решение 3** Через рекурсию (итеративный процесс).
```javascript
function fact(n) {
  function iter(count, acc) {
    if (count <= 1) {
      return acc;
    }

    return iter(count - 1, count * acc);
  }

  return iter(n, 1);
}

console.log(fact(2));
console.log(fact(0));
console.log(fact(5));
console.log(fact(-5));

```
