```JavaScript
function matrixMult(A, B) {
    return [
        [A[0][0] * B[0][0] + A[0][1] * B[1][0], A[0][0] * B[0][1] + A[0][1] * B[1][1]],
        [A[1][0] * B[0][0] + A[1][1] * B[1][0], A[1][0] * B[0][1] + A[1][1] * B[1][1]]
    ];
}

// –§—É–Ω–∫—Ü–∏—è –≤–æ–∑–≤–µ–¥–µ–Ω–∏—è –≤ —Å—Ç–µ–ø–µ–Ω—å —á–µ—Ä–µ–∑ –º–∞—Ç—Ä–∏—Ü—ã
function matrixPow(a, n) {
    // –ï–¥–∏–Ω–∏—á–Ω–∞—è –º–∞—Ç—Ä–∏—Ü–∞
    let result = [[1, 0], [0, 1]];
    // –ò—Å—Ö–æ–¥–Ω–∞—è –º–∞—Ç—Ä–∏—Ü–∞

    while (n > 0) {
        if (n % 2 === 1) {
            result = matrixMult(result, base);
        }
        base = matrixMult(base, base);
        n = Math.floor(n / 2);
    }
    
    return result[0][1]; // –í–æ–∑–≤—Ä–∞—â–∞–µ–º a^n
}

// –ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è

console.log(`7^1 = ${matrixPow(7, 1)}`); // 7
```
