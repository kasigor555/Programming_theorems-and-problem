–î–ª—è –≤—ã—á–∏—Å–ª–µ–Ω–∏—è –Ω–∞–º –ø–æ—Ç—Ä–µ–±—É–µ—Ç—Å—è —Ñ—É–Ω–∫—Ü–∏—è –¥–ª—è —Ä–∞—Å—á–µ—Ç–∞ —Ñ–∞–∫—Ç–æ—Ä–∏–∞–ª–∞ –∏–ª–∏ –∏—Ç–µ—Ä–∞—Ç–∏–≤–Ω—ã–π –ø–æ–¥—Ö–æ–¥, —á—Ç–æ–±—ã –∏–∑–±–µ–∂–∞—Ç—å –ø–µ—Ä–µ–∑–∞—Å—á–µ—Ç–∞ —Ñ–∞–∫—Ç–æ—Ä–∏–∞–ª–æ–≤. –ò—Ç–µ—Ä–∞—Ç–∏–≤–Ω—ã–π –ø–æ–¥—Ö–æ–¥ –±–æ–ª–µ–µ —ç—Ñ—Ñ–µ–∫—Ç–∏–≤–µ–Ω.

```JavaScript
/**
 * –í—ã—á–∏—Å–ª—è–µ—Ç –∑–Ω–∞—á–µ–Ω–∏–µ —á–∞—Å—Ç–∏—á–Ω–æ–π —Å—É–º–º—ã —Ä—è–¥–∞ e: S_n = 1/0! + 1/1! + ... + 1/n!

 * @returns {number} –°—É–º–º–∞ —Ä—è–¥–∞.

function calculatePartialSumOfE(n) {
    // –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –∫–æ—Ä—Ä–µ–∫—Ç–Ω–æ—Å—Ç—å –≤–≤–æ–¥–∞ (n –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–º —Ü–µ–ª—ã–º —á–∏—Å–ª–æ–º)

        throw new Error("N –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å –Ω–∞—Ç—É—Ä–∞–ª—å–Ω—ã–º —á–∏—Å–ª–æ–º (—Ü–µ–ª—ã–º, –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–º).");
    }

    // –î–ª—è n = 0 —Å—É–º–º–∞ —Ä–∞–≤–Ω–∞ 1/0! = 1
    if (n === 0) {
        return 1;
    }

    let sum = 0;
    // –§–∞–∫—Ç–æ—Ä–∏–∞–ª —Ç–µ–∫—É—â–µ–≥–æ —á–ª–µ–Ω–∞ k!
    // –ù–∞—á–∏–Ω–∞–µ–º —Å 1/0!
    let factorial = 1; 

    for (let k = 0; k <= n; k++) {
        // –ù–∞ –ø–µ—Ä–≤–æ–π –∏—Ç–µ—Ä–∞—Ü–∏–∏ (k=0): factorial = 1. –î–æ–±–∞–≤–ª—è–µ–º 1/1 = 1.
        // –ù–∞ –ø–æ—Å–ª–µ–¥—É—é—â–∏—Ö –∏—Ç–µ—Ä–∞—Ü–∏—è—Ö: factorial —Ö—Ä–∞–Ω–∏—Ç k!

            // –í—ã—á–∏—Å–ª—è–µ–º k! –Ω–∞ –æ—Å–Ω–æ–≤–µ (k-1)!, —É–º–Ω–æ–∂–∞—è –Ω–∞ k
            // factorial (–¥–ª—è k) = factorial (–¥–ª—è k-1) * k
            factorial *= k; 
        }

        // –î–æ–±–∞–≤–ª—è–µ–º 1/k! –∫ —Å—É–º–º–µ
        sum += (1 / factorial);
    }

    return sum;
}
```
